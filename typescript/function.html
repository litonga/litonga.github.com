    # 函数类型约束

    # 基础

    教程

    在js 中有两种常见的函数定义方式：函数声明和函数表达式 一个函数有输入输出，要约束一个函数的类型，其实就是约束函数的输入输出类型。

    # 函数声明类型定义：

    function sum(x:number, y:number):number{
	return x+y;
}

    # 函数表达式的定义：

    要注意函数类型定义中的=>与es6箭头函数中的=>的区别

    const a: (x: number, y: number) => number = (x, y) => {
  return x + y;
};

a(1, 3);

    # 接口定义函数的形状

    // 接口定义函数的形状5
interface Add {
  (x: number, y: number): number;
}
const add3: Add = (x, y) => x + y;
add3(1, 2);

    # 可选参数

    可选参数必须放在参数列表末尾

    function add3(x: number, y: number, z?: number): number {
  return x + y + (z || 0);
}

add3(1, 2, 3);

    # 参数的默认值

    TypeScript 会将添加了默认值的参数识别为可选参数 有默认值的可选参数可以放在任意位置

    function add4(x: number, y: number, z:number = 0): number {
  return x + y + z;  // z 被识别为可选参数
}

add4(1, 2, 3);

    # 剩余参数

    就是吧...rest 参数定义成数组类型

    function add5(...rest: number[]) {
  return rest.reduce((p, v) => p + v, 0);
}
add5(1, 2, 3, 4, 5, 6);

    # 重载

    function reload(x: number): number;
function reload(x: string): string;
function reload(x: number | string): number | string {
  return x;
}
reload(1);
reload("aaa");
